Sprites, Physics en Collisions

Deze module over Unity 2D gameontwikkeling bevatte de basis, sprites, physics, collisions en belighting.
Met behulp van javascript heb ik de objecten laten bewegen zoals te zien hier, je kan de diamand besturen met de pijltjes toetsen en de bal valt met zwaartekracht naar beneden.

Dit is een zelfstandig project dat niet verplicht is voor de opleiding, meer het lijkt mij intressant om meer te weten te komen over Game-Development. 

using UnityEngine;

public class GameManager : MonoBehaviour
{
    public float diamondSpeed = 5f;
    private GameObject square;
    private GameObject circle;
    private GameObject diamond;
    private GameObject isometricDiamond;

    void Start()
    {
        square = GameObject.Find("Square"); //start
        circle = GameObject.Find("Circle");
        diamond = GameObject.Find("Diamond");
        isometricDiamond = GameObject.Find("IsometricDiamond");
        if (square != null)
        {
            square.GetComponent().color = Color.red;
        }
        if (circle != null)
        {
            circle.transform.localScale = new Vector3(2f, 2f, 1f); 
        }
    }

    void Update()
    {
        if (diamond != null)
        {
            //beweging pijltjes
            float moveHorizontal = Input.GetAxis("Horizontal");
            diamond.transform.Translate(Vector3.right * moveHorizontal * diamondSpeed * Time.deltaTime);
        }

        if (isometricDiamond != null)
        {
            float moveHorizontal = Input.GetAxis("Horizontal");
            float moveVertical = Input.GetAxis("Vertical");
            isometricDiamond.transform.Translate(new Vector3(moveHorizontal, moveVertical, 0) * diamondSpeed * Time.deltaTime);
        }
    }

    void OnCollisionEnter2D(Collision2D collision)
    {
        if (collision.gameObject == square)
        {
            square.transform.Translate(Vector3.up * 5f);
        }
    }
}
