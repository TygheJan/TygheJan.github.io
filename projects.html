Wij hebben in groepjes van 4 een LEGO© Mindstorms robot geprogrameerd om een lijn te volgen met Python We hebben de sensors geconfigureerd en hiermee de robot een gekleurde lijn laten volgen.

        
        from mindstorms import MSHub, MotorPair, ColorSensor
        import time

        hub = MSHub()
        motors = MotorPair('A', 'C')
        color_sensor = ColorSensor('E')
        # motor_links = Motor('C')
        # motor_rechts = Motor('A')

        WHITE = "white"
        GREEN = "green"
        YELLOW = "yellow"
        BLACK = "black"

        last_color = None

        while True:
            color = color_sensor.get_color()
            print("Detected color value: " + str(color))

            if color == WHITE:
                print("Start detected (White)")
                motors.start(25, 25)
                time.sleep(1)
                motors.stop()

            elif color == GREEN:
                print("Green detected")
                motors.start(25, 1) # motors.start(20, 40)
                time.sleep(1)
                motors.stop()

            elif color == YELLOW:
                print("Yellow detected")
                motors.start(25, -25)
                time.sleep(1)
                motors.stop()

            elif color == BLACK:
                print("Black detected")
                motors.start(5, 5)
                time.sleep(1)
                motors.stop()
            else:
                print("Other detected")
